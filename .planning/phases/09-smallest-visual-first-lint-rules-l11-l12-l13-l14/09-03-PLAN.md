    ```javascript
    // ---------------------------------------------------------------------------
    // L13 вҖ” Double wrap (tree inside frame)
    // Severity: warn. A tree's indentation already conveys hierarchy; wrapping
    // it in a frame adds zero information at full border cost.
    // ---------------------------------------------------------------------------
    function L13_double_wrap(ast) {
      if (!ast || !ast.content) return [];
      const content = ast.content;
      if (!content.includes('в”Ң')) return [];
      if (!/[в”ңв””]в”Җв”Җ/.test(content)) return [];

      const lines = content.split('\n');
      const baseLineNum = ast.startLine;
      const issues = [];

      let li = 0;
      while (li < lines.length) {
        const top = lines[li];
        const topMatch = top.match(/^(\s*)в”Ңв”Җ+в”җ\s*$/);
        if (!topMatch) { li++; continue; }
        const indent = topMatch[1];

        let closeLi = -1;
        let hasTree = false;
        for (let lj = li + 1; lj < lines.length; lj++) {
          const next = lines[lj];
          const botMatch = next.match(/^(\s*)в””в”Җ+в”ҳ\s*$/);
          if (botMatch && botMatch[1] === indent) { closeLi = lj; break; }
          if (/[в”ңв””]в”Җв”Җ/.test(next)) hasTree = true;
        }
        if (closeLi === -1) { li++; continue; }

        if (hasTree) {
          issues.push(issue(
            'L13', 'warn', baseLineNum + li, indent.length + 1,
            `tree inside frame block вҖ” the tree already conveys hierarchy; drop the frame`,
            `Remove the surrounding в”Ңв”Җ...в”Җв”җ / в””в”Җ...в”Җв”ҳ borders; tree indentation suffices`
          ));
        }
        li = closeLi + 1;
      }
      return issues;
    }
    ```
