    ```javascript
    // STATE-MARKER allowlist (per D-09-04-03 in PLAN frontmatter).
    // A trailing token in an inner line counts as a "state" if it matches one
    // of these patterns. Cyrillic literals included; case-insensitive for ASCII.
    const STATE_MARKER_RE = /(?:← (?:готов|решение|заморожено|в работе|блок:[^│]+)|✓ \S+|✗ \S+|◐ \S+|⌛ \S+|→ \S+|done|pending|wip|ok|fail|wait)\s*$/i;

    /**
     * Convert a frame node (≤5 inner lines, no tree, no embedded table) to a
     * dot-leader list. Each inner line becomes one row.
     *   - "<label>  <state>" → "<label> <dots> <state>" (column-aligned)
     *   - free-form prose    → "- <content>"
     * Border lines stripped. Indentation preserved.
     * @param {{top:string, inner:string[], bottom:string, indent:string}} node
     * @returns {string}
     */
    function autofixFrameToDotLeader(node) {
      const indent = node.indent || '';
      const stripped = node.inner.map(line => unwrapInner(line, indent).trim());

      // Classify each row: pattern (label+state) or free-form
      const rows = stripped.map(content => {
        const stateMatch = content.match(STATE_MARKER_RE);
        if (!stateMatch) {
          return { kind: 'bullet', content };
        }
        const state = stateMatch[0].trim();
        // Label = everything before the state, trimmed. May contain dots from
        // a prior dot-leader render — strip trailing dots+spaces.
        let label = content.slice(0, stateMatch.index).replace(/[\s.]+$/, '');
        return { kind: 'pattern', label, state };
      });

      // If ANY row is bullet, emit ALL as bullets (mixed mode is too noisy).
      // Otherwise compute column width and emit dot-leader rows.
      const allPattern = rows.every(r => r.kind === 'pattern');
      if (!allPattern) {
        return rows.map(r => {
          const text = r.kind === 'pattern' ? `${r.label} ${r.state}` : r.content;
          return indent + '- ' + text;
        }).join('\n');
      }

      const labelMax = Math.max(...rows.map(r => visualWidth(r.label)));
      const stateMax = Math.max(...rows.map(r => visualWidth(r.state)));
      // Per D-09-04-01: auto-detect, cap at 80 cols (account for indent).
      const indentW = visualWidth(indent);
      const wantW = labelMax + 4 + stateMax;
      const W = Math.min(80 - indentW, wantW);

      return rows.map(r => {
        const labelW = visualWidth(r.label);
        const stateW = visualWidth(r.state);
        const dotsCount = Math.max(3, W - labelW - stateW - 2); // -2 for spaces around dots
        const dots = '.'.repeat(dotsCount);
        return indent + r.label + ' ' + dots + ' ' + r.state;
      }).join('\n');
    }
    ```

    ```javascript
      });

      function stripFence(text) {
        return text.replace(/^```[a-z]*\n?/gm, '').replace(/```\n?$/gm, '');
      }

      for (const fx of fixtures) {
        it(`autofix shape matches: ${fx.name}`, () => {
          const out = autofix(fx.input);
          const shape = fx.expected_after_autofix_shape;
          const body = stripFence(out).replace(/\n+$/, '');
          const lines = body.split('\n').filter(l => l.length > 0);

          // 1. no_frame_chars
          if (shape.no_frame_chars) {
            assert.doesNotMatch(out, /[┌┐└┘]/, `frame corners must be removed in "${fx.name}"`);
            // │ allowed only if mode is no_change AND input had │ (it should not for dot-leader inputs)
          }

          // 2. row_count
          if (typeof shape.row_count === 'number') {
            assert.equal(lines.length, shape.row_count,
              `expected ${shape.row_count} rows, got ${lines.length} in "${fx.name}":\n${body}`);
          }

          // 3. mode-specific shape
          if (shape.mode === 'dot_leader') {
            // Every row: "<label> <dots≥3> <state>"
            for (const ln of lines) {
              assert.match(ln, /^\S.* \.{3,} \S/, `row "${ln}" must be dot-leader shape`);
            }
            if (shape.aligned_dots) {
              const dotCounts = lines.map(ln => (ln.match(/\.+/) || [''])[0].length);
              assert.equal(new Set(dotCounts).size, 1,
                `dot-counts must be equal across rows in "${fx.name}", got ${dotCounts.join(',')}`);
            }
          } else if (shape.mode === 'bullet') {
            for (const ln of lines) {
              assert.match(ln, /^\s*-\s+\S/, `row "${ln}" must be bullet`);
            }
          } else if (shape.mode === 'no_change') {
            // Idempotency: output equals input
            assert.equal(out, fx.input, `no_change mode: output must equal input for "${fx.name}"`);
          }

          // 4. preserves list (case-sensitive substring)
          for (const token of (shape.preserves || [])) {
            assert.ok(out.includes(token),
              `expected to preserve "${token}" in output of "${fx.name}":\n${out}`);
          }
        });

        it(`autofix is idempotent: ${fx.name}`, () => {
          const once = autofix(fx.input);
          const twice = autofix(once);
          assert.equal(twice, once, `second pass must be no-op for "${fx.name}"`);
        });
      }
    ```
