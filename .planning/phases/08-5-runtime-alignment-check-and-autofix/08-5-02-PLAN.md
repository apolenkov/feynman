### Task 2 — GREEN: Implement
In `lib/lint/rules.js`, after L09 export, add:

```js
const PKG_TOKENS = (() => {
  try {
    const pkg = require('../../package.json');
    return new Set([
      pkg.name, pkg.name && pkg.name.replace(/^@[^/]+\//, ''),
      ...(pkg.keywords || []),
      ...Object.keys(pkg.bin || {}),
    ].filter(Boolean));
  } catch { return new Set(); }
})();

function L10_mixed_script(text) {
  const violations = [];
  const lines = text.split('\n');
  const tokenRe = /[\p{L}\p{N}_-]+/gu;
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];
    let m;
    while ((m = tokenRe.exec(line)) !== null) {
      const token = m[0];
      // Whitelist
      if (PKG_TOKENS.has(token)) continue;
      // Kebab-case (any token containing '-')
      if (token.includes('-')) continue;
      // Numeric suffix (foo123, foo-123)
      if (/^[a-zA-Z]+\d+$/.test(token)) continue;
      // Mixed script check
      if (/[А-я]/.test(token) && /[A-Za-z]/.test(token)) {
        violations.push({ line: i + 1, column: m.index + 1, token, severity: 'warn', rule: 'L10' });
      }
    }
  }
  return violations;
}

module.exports.L10_mixed_script = L10_mixed_script;
```

Wire into the main lint dispatcher (likely `lib/lint/index.js`): include L10 in the
rule list, but with severity `warn` so it does NOT change exit code.

Run `npm test`. L10 fixtures pass; existing 245 stay green.

## Risk

- The Cyrillic block `[А-я]` covers Russian but misses other Cyrillic scripts (Bulgarian,
  Serbian, Macedonian); for v1.0 this is sufficient — feynman is dev-tool, RU+EN is the
  expected mix. Wider script coverage = post-8.5.
- `\p{L}` Unicode property regex requires `u` flag (already used). Node ≥ 18 supports it
  natively, no shim needed.
